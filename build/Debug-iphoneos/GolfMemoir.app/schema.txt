CREATE TABLE app ('pk' INTEGER PRIMARY KEY, 'appVersion' INTEGER, 'privacyAccepted' Integer);
CREATE TABLE course ('pk' INTEGER PRIMARY KEY, 'courseName' CHAR(255), 'courseData' CHAR(48), 'lastDate' REAL, 'holeNumber' INTEGER , 'sr' INTEGER, 'cr' INTEGER, 'whiteSlope' INTEGER, 'blackSlope' INTEGER, 'blueSlope' INTEGER, 'orangeSlope' INTEGER, 'extraSlope' INTEGER, 'whiteRate' DOUBLE, 'blackRate' DOUBLE, 'blueRate' DOUBLE, 'orangeRate' DOUBLE, 'extraRate' DOUBLE, 'bool1' INTEGER, 'bool2' INTEGER);
CREATE TABLE coursehole ('courseID' INTEGER, 'holeID' INTEGER, 'whitePar' INTEGER, 'whiteYard' INTEGER, 'blackYard' INTEGER, 'blueYard' INTEGER, 'orangeYard' INTEGER, 'hcp' INTEGER, 'longitude' DOUBLE, 'latitude' DOUBLE, 'extraYard' INTEGER,  'bool1' INTEGER, 'bool2' INTEGER, 'holeType' INTEGER, 'teeLongitude' DOUBLE, 'teeLatitude' DOUBLE, 'frontLongitude' DOUBLE, 'frontLatitude' DOUBLE, 'backLongitude' DOUBLE, 'backLatitude' DOUBLE);
CREATE TABLE hole ('scoreID' INTEGER, 'holeID' INTEGER, 'strokeNum' INTEGER, 'strokeNum2' INTEGER, 'strokeNum3' INTEGER, 'strokeNum4' INTEGER);
CREATE TABLE score ('pk' INTEGER PRIMARY KEY, 'courseID' INTEGER, 'playDate' REAL, 'player2Name' CHAR(48), 'player3Name' CHAR(48), 'player4Name' CHAR(48), 'holeNumber' INTEGER, 'status' INTEGER, 'hcp' INTEGER, 'hcp2', INTEGER, 'hcp3' INTEGER, 'hcp4' INTEGER);
CREATE TABLE stroke ('scoreID' INTEGER, 'holeID' INTEGER, 'strokeNum' INTEGER, 'distance' INTEGER, 'longitude' DOUBLE, 'latitude' DOUBLE, 'putt' INTEGER);
